.Tools[]
  | [
      .Name,
      (.Items
      | sort_by(.Compliancy)
      | last
      | [["linuxx86-64", .Version, .Urls."linuxx86-64", .Urls."linuxx86-64-sha1"], ["linuxarm64", .Version, .Urls."linuxarm64", .Urls."linuxarm64-sha1"]])
    ]
| first as $name
| nth(2) as $version
| reduce last[] as $arch ([]; . + [[$name,$arch]])
| map(flatten)

depends=()
noextract=(
    ${source_x86_64[@]}
    ${source_aarch64[@]}
)

package() {
    cd "$srcdir"
    case "$arch" in
       x86_64) _binfile="${_binname}-${pkgver}-linuxx86-64" ;;
       aarch64) _binfile="${_binname}-${pkgver}-linuxarm64" ;;
    esac
    mv "${_binfile}" "${_binname}"
    # TODO: ln -sf "${_binfile}" "${_binname}" and move both?
    install -Dm0755 -t "$pkgdir/usr/bin" "$srcdir/${_binname}"
}
